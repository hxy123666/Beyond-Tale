3. 在JavaScript部分添加音乐控制功能（在结束标签前添加） javascript // =============== 背景音乐控制功能 =============== const audio = document.getElementById('background-music'); const toggleMusicBtn = document.getElementById('toggle-music'); const volumeControl = document.getElementById('volume'); // 尝试自动播放音乐 function tryAutoplay() { // 由于浏览器策略，需要用户交互后才能自动播放 const promise = audio.play(); if (promise !== undefined) { promise.then(() => { // 自动播放成功 toggleMusicBtn.textContent = '🔊'; }).catch(error => { // 自动播放失败，需要用户手动启动 toggleMusicBtn.textContent = '🔇'; console.log('自动播放受阻，需要用户交互'); }); } } // 切换播放/暂停 function toggleMusic() { if (audio.paused) { audio.play(); toggleMusicBtn.textContent = '🔊'; } else { audio.pause(); toggleMusicBtn.textContent = '🔇'; } } // 设置音量 function setVolume() { audio.volume = volumeControl.value; } // 事件监听器 document.addEventListener('click', function() { // 用户首次交互后尝试自动播放 if (audio.paused) { tryAutoplay(); } }, { once: true }); toggleMusicBtn.addEventListener('click', toggleMusic); volumeControl.addEventListener('input', setVolume); // 初始化音量 audio.volume = volumeControl.value; // =============== 音乐控制功能结束 ===============
